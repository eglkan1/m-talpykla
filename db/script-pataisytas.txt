CREATE TABLE TEACHERS(
    id INT,
    NAME VARCHAR(255),
    surname VARCHAR(255),
    username VARCHAR(255),
    PASSWORD VARCHAR(255),
    email VARCHAR(255),
    school VARCHAR(255),
    PRIMARY KEY(id)
); CREATE TABLE USERS(id INT, PRIMARY KEY(id)); CREATE TABLE FOLDERS(
    id INT,
    title VARCHAR(255),
    PASSWORD VARCHAR(255),
    public BOOLEAN,
    fk_TEACHERid INT(11),
    PRIMARY KEY(id),
    CONSTRAINT manage FOREIGN KEY(fk_TEACHERid) REFERENCES TEACHERS(id)
); CREATE TABLE FILES(
    id INT,
    title VARCHAR(255),
    FILE VARCHAR(255),
    public BOOLEAN,
    description VARCHAR(255),
    fk_FOLDERid INT(11) NOT NULL,
    PRIMARY KEY(id),
    CONSTRAINT had FOREIGN KEY(fk_FOLDERid) REFERENCES FOLDERS(id)
); CREATE TABLE TESTS(
    title VARCHAR(255),
    duration INT,
    public BOOLEAN,
    id INT,
    fk_FOLDERid INT(11) NOT NULL,
    PRIMARY KEY(id),
    CONSTRAINT include FOREIGN KEY(fk_FOLDERid) REFERENCES FOLDERS(id)
); CREATE TABLE QUESTIONS(
    question VARCHAR(255),
    id INT(11),
    image VARCHAR(255),
    TYPE CHAR(0),
    fk_TESTid INT(11) NOT NULL,
    PRIMARY KEY(id),
    CONSTRAINT has FOREIGN KEY(fk_TESTid) REFERENCES TESTS(id)
); CREATE TABLE RESPONSES(
    isRight BOOLEAN,
    id_RESPONSE INTEGER,
    fk_USERid INT(11) NOT NULL,
    fk_QUESTIONid INT(11) NOT NULL,
    fk_TESTid INT(11) NOT NULL,
    PRIMARY KEY(id_RESPONSE),
    CONSTRAINT response FOREIGN KEY(fk_USERid) REFERENCES USERS(id),
    CONSTRAINT question FOREIGN KEY(fk_QUESTIONid) REFERENCES QUESTIONS(id),
    CONSTRAINT belong FOREIGN KEY(fk_TESTid) REFERENCES TESTS(id)
); CREATE TABLE ANSWERS(
    answer VARCHAR(255),
    isRightChoice BOOLEAN,
    id INT,
    fk_QUESTIONid INT(11) NOT NULL,
    fk_RESPONSEid_RESPONSE INTEGER NOT NULL,
    PRIMARY KEY(id),
    CONSTRAINT own FOREIGN KEY(fk_QUESTIONid) REFERENCES QUESTIONS(id),
    CONSTRAINT answer FOREIGN KEY(fk_RESPONSEid_RESPONSE) REFERENCES RESPONSES(id_RESPONSE)
);
